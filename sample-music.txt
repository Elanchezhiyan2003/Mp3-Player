let play = document.getElementById("play");
let audio = new Audio("sample.mp3");  
let isPlaying = false;  

function Music() {
    if (isPlaying) {
        audio.pause();
        // playButton.src = "images/play-button.png";  
    } else {
        audio.play(); 
        // playButton.src = "images/pause.png";  
    }
    isPlaying = !isPlaying;  
}

play.addEventListener("click", Music);
